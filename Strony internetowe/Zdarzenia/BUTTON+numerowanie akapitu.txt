    <script>
        let button = document.querySelector("#Przycisk");
        let counter = 1;

        button.addEventListener("mouseover", mover);
        button.addEventListener("click", mclick);
        function mover(){
            document.querySelector("#Demo").innerHTML +="over <br>";
        }
        function mclick(){
            
            let p=document.createElement("p");
            p.textContent="Nowy akapit po kliknieciu" + counter;
            document.querySelector("#Demo").appendChild(p);
            counter++;
        }

    </script>
